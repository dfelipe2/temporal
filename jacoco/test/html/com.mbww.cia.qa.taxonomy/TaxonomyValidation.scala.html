package com.mbww.cia.qa.taxonomy

import com.mbww.cia.common._
import com.mbww.cia.qa.bitmap.BitmapFileValidation
import com.mbww.cia.qa.dataProvider.DataProviderValidation
import com.mbww.cia.qa.dataSet.DataSetValidation
import com.mbww.cia.qa.displayName.DisplayNameValidation
import com.mbww.cia.qa.idMapping.IdMappingValidation
import com.mbww.cia.qa.matchingPartner.MatchingPartnerValidation
import org.apache.log4j.Logger
import org.json.JSONObject
class TaxonomyValidation extends Module {
  private[this] val logger = Logger.getLogger(getClass().getName())
  override def run(args: JSONObject): Unit = {

    val marketName = args.getString("marketName")
    val bucket = args.getString("bucket")
    val dataProviderName = args.getString("dataProviderName")
    val matchingPartnerName = args.getString("matchingPartnerName")
    val thread_count = args.getOrElse[Int]("thread_count", 1)
    val bitmapFileValidation = new BitmapFileValidation()
    val dataProvider = new DataProviderValidation()
    val dataSetValidation = new DataSetValidation()
    val displayNameValidation = new DisplayNameValidation()
    val idMappingValidation = new IdMappingValidation()
    val matchingPartnerValidation = new MatchingPartnerValidation()
    if(!bitmapFileValidation.validation(
            marketName,
            bucket,
            dataProviderName,
            matchingPartnerName,
            thread_count)){
      logger.info("########### Validation fail for bitmapFileValidation ########### ")
      throw new RuntimeException}

   if(!dataProvider.validation(
            marketName,
            bucket,
            dataProviderName,
            matchingPartnerName,
            thread_count)){
     logger.info("########### Validation fail for dataProvider ########### ")
     throw new RuntimeException}

     if(!dataSetValidation.validation(
            marketName,
            bucket,
            dataProviderName,
            matchingPartnerName,
            thread_count)){
       logger.info("########### Validation fail for dataSetValidation ########### ")
     throw new RuntimeException }

   if(!displayNameValidation.validation(
            marketName,
            bucket,
            dataProviderName,
            matchingPartnerName,
            thread_count)){
      logger.info("########### Validation fail for bitmapFileValidation ########### ")
      throw new RuntimeException }

    if(!idMappingValidation.validation(
            marketName,
            bucket,
            dataProviderName,
            matchingPartnerName,
            thread_count)){
      logger.info("########### Validation fail for idMappingValidation ########### ")
      throw new RuntimeException}

    if(!matchingPartnerValidation.validation(
            marketName,
            bucket,
            dataProviderName,
            matchingPartnerName,
            thread_count)){
      logger.info("########### Validation fail for matchingPartnerValidation ########### ")
      throw new RuntimeException}

    }
}
