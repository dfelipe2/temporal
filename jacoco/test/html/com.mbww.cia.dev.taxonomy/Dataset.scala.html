package com.mbww.cia.dev.taxonomy

import com.mbww.cia.common.{Module, _}
import org.json.JSONObject

import scala.collection.Seq

object Dataset {

  val DATA_PROVIDER = "Data Provider"
  val DATA_SOURCE = "Data Source"
  val DISPLAY_NAME = "Display Name"
  val DATASET_TYPE = "Dataset Type"
  val DATA_SOURCE_BITMAP_ROOT_PATH = "Data Source Bitmap Root Path"
  val DATA_SOURCE_CUSTOM_ATTRIBTUE_ROOT_PATH = "Data Source Custom Attribute Root Path"
  val DATA_SOURCE_TAXONOMY_ROOT_PATH = "Data Source Taxonomy Root Path"
  val TIMELINE = "TimeLine"
  val VERSION = "Version"
  val MINOR_VERSION = "Minor Version"
  val MBID_BITMAP_ROOT_FOLDER = "MBID Bitmap Root Folder"
  val HHID_BITMAP_ROOT_FOLDER = "HHID Bitmap Root Folder"
  val DATASET_DISPLAY_FILE = "Dataset Display File"
  val DTASET_BITMAP_FILE = "Dataset Bitmap File"

  val ID_INDEX = 0

  val CSV_HEADER = Seq(
    DATA_PROVIDER,
    DATA_SOURCE,
    DISPLAY_NAME,
    DATASET_TYPE,
    DATA_SOURCE_BITMAP_ROOT_PATH,
    DATA_SOURCE_CUSTOM_ATTRIBTUE_ROOT_PATH,
    DATA_SOURCE_TAXONOMY_ROOT_PATH,
    TIMELINE,
    VERSION,
    MINOR_VERSION,
    MBID_BITMAP_ROOT_FOLDER,
    HHID_BITMAP_ROOT_FOLDER,
    DATASET_DISPLAY_FILE,
    DTASET_BITMAP_FILE)
}

class Dataset extends Module {

  def generate(inputFile: String, newRow: Seq[String]) = {

    updateCSVFile(
      inputFile,
      Dataset.CSV_HEADER,
      Dataset.ID_INDEX,
      newRow)
  }

  override def run(args: JSONObject): Unit = {

    val inputFile = args.getString("input_file")
    val values = Dataset.CSV_HEADER.map(c => args.getString(c).trim)

    generate(inputFile, values)
  }
}
