package com.mbww.cia.core.logging

}

trait CIALogger {

  def info(msg: String, attrs: MessageAttribute*): Unit

  def debug(msg: String, attrs: MessageAttribute*): Unit

  def error(msg: String, attrs: MessageAttribute*): Unit

}
