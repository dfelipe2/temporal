package com.mbww.cia.core.pipeline

import java.util.concurrent.Callable

import com.mbww.cia.common.Module
import org.json.JSONObject

import scala.util.Try

class Step(val clazz: String,
           val name: String,
           val num: Int,
           val args: JSONObject) extends Callable[Try[Unit]] {

  // this can throw an exception. Let's fail fast
  val module = Class.forName(clazz)

  def run(): Unit = {

    module
      .getDeclaredConstructor()
      .newInstance()
      .asInstanceOf[Module]
      .run(args)
  }

  override def call(): Try[Unit] = {
    Try(run())
  }
}
