  def reportAcxiomUK(acxiom_path: String,
                     desired_columns_path: String,
                     acxiom_keys_path: String,
                     output_path: String
                    ): Unit = {

    var acxiomInput = spark.read.parquet(acxiom_path)
    val uniqueKeys = spark.read.parquet(acxiom_keys_path)
    val columnsDf = spark.read.option("delimiter","|").option("header","true").csv(desired_columns_path)
    val uniqueKeys_withRowNum=uniqueKeys.withColumn("rownum", functions.monotonically_increasing_id())
    val r = scala.util.Random
    val rangeIndex=for (i <- 1 to 10) yield r.nextInt(5000)

    for (data <- rangeIndex) {
      println("rownum under test is :" + data)
      val filteredDataFrame=uniqueKeys_withRowNum.filter(col("rownum").equalTo(s"${data}")).drop("rownum")

      val numericIdExpected = filteredDataFrame.select("numericId")
      log.info(s"numeric Id Expected" + numericIdExpected)

      val acxiomDf = filteredDataFrame.join(acxiomInput, col("stringId") === col("mbid_v2"), "inner")
      acxiomDf.createOrReplaceTempView("acxiom_table")

      val columnsMap = columnsDf.collect()

      val rowPrefix = ",cast((case when "
      val rowSufix = " then 1 else 0 end) as tinyint) as "
      var rowsSQL = ""

      for (col <- columnsMap) {
        val originColName = col.getAs[String]("origin_column")
        val oneValues = col.getAs[String]("one_values")
        val aliasColName = col.getAs[String]("alias_column_name")
        val caseSting = rowPrefix + originColName + " in " + oneValues + rowSufix + aliasColName
        rowsSQL += caseSting
      }

      val sqlPrefix = "SELECT CAST(numericId as bigint) as numericId "
      val sqlSufix = " FROM acxiom_table"
      val fullSQL = sqlPrefix + rowsSQL + sqlSufix

      val expectedProfileReport = spark.sql(fullSQL)
      val expectedCount = expectedProfileReport.count()

      val b = numericIdExpected.take(1).toArray
      val c = b(0).mkString
      val actualProfileReport = spark.read.load(output_path)
      val filteredactualProfileReport = actualProfileReport.filter(s"numericId='${c}'")
      val actualCount = filteredactualProfileReport.count()

      val datadiff = expectedProfileReport.except(filteredactualProfileReport)
      val difference1 = datadiff.count
      log.info(s"difference of expected profile report and actual profile report " + difference1)

      val datadiff1 = filteredactualProfileReport.except(expectedProfileReport)
      val difference2 = datadiff1.count
      log.info(s"difference of actual profile report and expected profile report " + difference2)

      if (!(expectedCount == actualCount)) {
        throw new RuntimeException(
          s"Actual  count ${expectedCount} is not same as Expected count ${actualCount} ")

      }
      log.info(s"Actual  count ${expectedCount} is  same as Expected count ${actualCount} ")


      if (!(difference1 == difference2)) {
        throw new RuntimeException(
          s"difference of expected profile report and actual profile report  ${difference1}  is not same as difference of actual profile report and expected profile report ${difference2} ")
      }
      log.info(s"difference of expected profile report and actual profile report ${difference1}is same as  difference of actual profile report and expected profile report ${difference2} ")
    }
    true
  }
